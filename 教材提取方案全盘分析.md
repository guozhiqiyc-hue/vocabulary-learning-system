## 格式对比分析

### PDF格式特点

**优点**:
- ✅ **保留原始排版** - 与纸质教材完全一致
- ✅ **图文清晰** - 高质量扫描或排版
- ✅ **跨平台兼容** - 任何设备都能查看
- ✅ **成熟提取工具** - pdfplumber、PyMuPDF等

**缺点**:
- ⚠️ **提取复杂度中等** - 需要处理PDF结构
- ⚠️ **可能需要OCR** - 扫描版PDF需要图像识别
- ⚠️ **文件较大** - 七上51MB，七下68MB

**数据提取难度**: ⭐⭐ (2/5)

### DOCX格式特点

**优点**:
- ✅ **结构化数据** - 直接包含文本、样式信息
- ✅ **提取最简单** - 标准XML格式，易解析
- ✅ **无需OCR** - 纯文本，100%准确
- ✅ **文件较小** - 17MB vs PDF的51-68MB
- ✅ **包含元数据** - 标题、样式、格式信息

**缺点**:
- ⚠️ **排版可能丢失** - 与PDF相比可能略有差异
- ⚠️ **图片分离** - 图片和文本分开存储

**数据提取难度**: ⭐ (1/5)

### 格式选择建议

| 场景 | 推荐格式 | 理由 |
|------|---------|------|
| **快速提取** | DOCX | 直接读取XML，无需复杂处理 |
| **完整内容** | PDF | 保留所有图片和排版 |
| **词汇提取** | DOCX | 文本清晰，识别准确 |
| **混合方案** | DOCX为主+PDF补充 | DOCX提取词汇，PDF补充图片 |

```python
# complete_textbook_extractor.py
import json
import re
from pathlib import Path
from typing import Dict, List, Any
import pdfplumber
from docx import Document

class TextbookExtractor:
    """教材完整内容提取器"""

    def __init__(self, filepath: Path):
        self.filepath = filepath
        self.suffix = filepath.suffix.lower()
        self.data = {
            "metadata": {
                "filename": filepath.name,
                "grade": self._detect_grade(),
                "semester": self._detect_semester(),
                "publisher": "译林版"
            },
            "units": [],
            "vocabulary": [],
            "dialogues": [],
            "readings": [],
            "grammar": []
        }

    def _detect_grade(self):
        name = self.filepath.name
        if '七' in name: return '七年级'
        if '八' in name: return '八年级'
        return '未知'

    def _detect_semester(self):
        name = self.filepath.name
        if '上' in name: return '上册'
        if '下' in name: return '下册'
        return '未知'

    def extract_from_pdf(self):
        """从PDF提取"""
        with pdfplumber.open(self.filepath) as pdf:
            for i, page in enumerate(pdf.pages):
                text = page.extract_text()
                if not text:
                    continue

                # 检测单元
                self._detect_unit(text, i + 1)

                # 提取词汇
                self._extract_vocab(text)

                # 提取对话
                self._extract_dialogue(text, i + 1)

                # 提取阅读
                self._extract_reading(text, i + 1)

                # 提取语法
                self._extract_grammar(text, i + 1)

    def extract_from_docx(self):
        """从DOCX提取"""
        doc = Document(self.filepath)
        full_text = []

        for para in doc.paragraphs:
            full_text.append(para.text)

        text = "\n".join(full_text)

        # 检测单元
        self._detect_unit(text, 0)

        # 提取词汇
        self._extract_vocab(text)

        # ... 其他提取

    def _detect_unit(self, text, page):
        """检测单元标题"""
        patterns = [
            r'Unit\s*\d+[:：]?\s*[^\n]+',
            r'第?\s*[一二三四五六七八九十\d]+\s*单元'
        ]

        for pattern in patterns:
            matches = re.findall(pattern, text, re.MULTILINE | re.IGNORECASE)
            for match in matches:
                if match.strip() not in [u['title'] for u in self.data['units']]:
                    self.data['units'].append({
                        'title': match.strip(),
                        'page': page,
                        'vocabulary': [],
                        'dialogues': [],
                        'readings': []
                    })

    def _extract_vocab(self, text):
        """提取词汇"""
        # 匹配: word n. 释义
        pattern = r'\b([a-zA-Z]{3,})\s+([n\.|v\.|adj\.|adv\.])\s+([^\n]+)'

        matches = re.findall(pattern, text)
        for word, pos, meaning in matches:
            self.data['vocabulary'].append({
                'word': word,
                'pos': pos,
                'meaning': meaning.strip()
            })

    def _extract_dialogue(self, text, page):
        """提取对话"""
        lines = text.split('\n')
        dialogue = []
        speakers = set()

        for line in lines:
            # 匹配: Speaker: content
            match = re.match(r'^([A-Za-z]+)[:：]\s*(.+)$', line)
            if match:
                speaker, content = match.groups()
                dialogue.append({
                    'speaker': speaker,
                    'content': content.strip()
                })
                speakers.add(speaker)

        if len(dialogue) > 3:  # 至少3轮对话
            self.data['dialogues'].append({
                'page': page,
                'speakers': list(speakers),
                'lines': dialogue
            })

    def _extract_reading(self, text, page):
        """提取阅读文章"""
        # 简化版: 提取长段落
        paragraphs = text.split('\n\n')
        for para in paragraphs:
            if len(para.split()) > 50:  # 超过50词
                self.data['readings'].append({
                    'page': page,
                    'content': para.strip(),
                    'word_count': len(para.split())
                })
                break  # 每页只取一个

    def _extract_grammar(self, text, page):
        """提取语法点"""
        if 'Grammar' in text or '语法' in text:
            self.data['grammar'].append({
                'page': page,
                'content': text[:500]  # 取前500字符
            })

    def save(self, output_path: Path):
        """保存数据"""
        output_path.parent.mkdir(parents=True, exist_ok=True)

        with open(output_path, 'w', encoding='utf-8') as f:
            json.dump(self.data, f, ensure_ascii=False, indent=2)

        print(f"✓ 数据已保存: {output_path}")
        print(f"  单元: {len(self.data['units'])}")
        print(f"  词汇: {len(self.data['vocabulary'])}")
        print(f"  对话: {len(self.data['dialogues'])}")
        print(f"  阅读: {len(self.data['readings'])}")

    def extract(self):
        """根据文件类型提取"""
        if self.suffix == '.pdf':
            self.extract_from_pdf()
        elif self.suffix == '.docx':
            self.extract_from_docx()
        else:
            raise ValueError(f"不支持的格式: {self.suffix}")
```

```python
def convert_to_system_format(full_data_path):
    """将完整数据转换为system格式"""

    with open(full_data_path, 'r', encoding='utf-8') as f:
        data = json.load(f)

    system_data = {
        "source": data['metadata']['filename'],
        "total": 0,
        "cards": []
    }

    for vocab in data['vocabulary']:
        card = {
            "id": f"{data['metadata']['grade']}_{len(system_data['cards']) + 1:03d}",

            "meaning": vocab['meaning'],

            "keywords": [vocab['word']],

            "type": "vocabulary"
        }

    system_data['total'] = len(system_data['cards'])

    with open(output, 'w', encoding='utf-8') as f:
        json.dump(system_data, f, ensure_ascii=False, indent=2)
```

```python
def extract_from_docx(docx_path):

    doc = Document(docx_path)
    current_unit = ""

                "type": "vocabulary"
            }
            cards.append(card)

    return {
        "source": Path(docx_path).name,
        "total": len(cards),
        "cards": cards
    }
```

```python
# add_examples.py
import json
from pathlib import Path
import random

# 例句模板库
example_templates = {
    "be good at": [
        "I **am good at** playing basketball.",
        "She **is good at** mathematics.",
        "They **are good at** English."
    ],
    # ... 更多短语
}

def add_examples_from_templates(data_path):
    """从模板库添加例句"""

    with open(data_path, 'r', encoding='utf-8') as f:
        data = json.load(f)

    for card in data['cards']:
        phrase = card['phrase'].lower()

        # 如果有模板，随机选择一个
        if phrase in example_templates:
            examples = example_templates[phrase]
            card['example'] = random.choice(examples)

    # 保存
    output = data_path.parent / f"{data_path.stem}_with_examples.json"
    with open(output, 'w', encoding='utf-8') as f:
        json.dump(data, f, ensure_ascii=False, indent=2)

    print(f"✓ 已添加例句: {output}")

# 为所有数据添加例句
for name in ['七上', '七下', '八上', '八下']:
    expanded = Path(f"提取数据/{name}_expanded.json")
    if expanded.exists():
        add_examples_from_templates(expanded)
```
